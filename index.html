void readRemoteControl() { // Check if there's a remote relay command if (Firebase.RTDB.getBool(&fbdo, "/solar_system/remote_relay")) { if (fbdo.dataType() == "boolean") { bool remoteRelay = fbdo.boolData(); // Only change if different if (remoteRelay != relayState) { Serial.printf("Remote relay command received: %s\n", remoteRelay ? "ON" : "OFF"); // Update relay relayState = remoteRelay; digitalWrite(RELAY_PIN, relayState ? HIGH : LOW); // Update Firebase with new state Firebase.RTDB.setBool(&fbdo, "/solar_system/relay_state", relayState); // Provide feedback beepBuzzer(1); // Clear the remote command (prevents repeating) Firebase.RTDB.setBool(&fbdo, "/solar_system/remote_relay", false); } } } // Check for reset command if (Firebase.RTDB.getBool(&fbdo, "/solar_system/reset")) { if (fbdo.dataType() == "boolean" && fbdo.boolData()) { Serial.println("Remote reset command received!"); // Reset relay to OFF relayState = false; digitalWrite(RELAY_PIN, LOW); // Update Firebase Firebase.RTDB.setBool(&fbdo, "/solar_system/relay_state", false); // Beep to confirm beepBuzzer(3); // Clear reset command Firebase.RTDB.setBool(&fbdo, "/solar_system/reset", false); } } }
